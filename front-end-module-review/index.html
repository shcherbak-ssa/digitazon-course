function getAvarageTemperature(weekList = []) {
  if (weekList.length === 0) return 0;

  const temperatureSum = weekList.reduce((sum, { temperature }) => {
    return sum + temperature;
  }, 0);

  return temperatureSum / weekList.length
}

              function getMaxTemperature(weekList = []) {
  if (weekList.length === 0) return 0;

  const temperatures = weekList.map(({ temperature }) => temperature);

  return Math.max(...temperatures);
}

function getMinTemperature(weekList = []) {
  if (weekList.length === 0) return 0;

  const temperatures = weekList.map(({ temperature }) => temperature);

  return Math.min(...temperatures);
}
